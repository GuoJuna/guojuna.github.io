# vuepress-plugin-meting音乐插件

  • 安装
yarn add vuepress-plugin-meting -D
# or use npm
npm i vuepress-plugin-meting -D
  • 配置
## .vuepress/config.js
plugins: [
      ["vuepress-plugin-boxx"],
      ['meting', {
        //metingApi: "https://meting.sigure.xyz/api/music",
        meting: {
          // "netease" | "tencent" | "xiami"
          server: "tencent",
          // 读取歌单
          type: "playlist",
          mid: "2006719587",
        },
        // 不配置该项的话不会出现全局播放器
        aplayer: {
          // 吸底模式
          fixed: true,
          mini: false,
          // 自动播放
          autoplay: true,
          // 歌曲栏折叠
          listFolded:true,
          // 颜色
          theme: '#000000',
          // 播放顺序为随机
          order: 'random',
          // 初始音量
          volume: 0.1,
          // 关闭歌词显示
          lrcType: 0
        },
        mobile :{
          // 手机端去掉cover图
          cover: false,
        }
      }]
  ]
